# S3

```yaml
parameters:
    aws.region.west: 'eu-west-1'

services:
    
    aws.credentials:
        class: Aws\Credentials\Credentials
        arguments: ['%env(AWS_ACCESS_KEY_ID)%', '%env(AWS_SECRET_ACCESS_KEY)%']
    
    aws.s3.client.west:
        factory: ['Landingi\AwsBundle\Aws\S3\ClientFactory', 'build']
        arguments:
            - '@aws.credentials'
            - '%aws.region.west%'

    Landingi\AwsBundle\Aws\S3Storage:
        class: Landingi\AwsBundle\Aws\S3Storage
        arguments: ['@aws.s3.client.west']
```
